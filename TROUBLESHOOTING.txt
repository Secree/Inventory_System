Problem 1: "Permission denied"
Fix:
  chmod +x main.py
  chmod +x launch.sh

Problem 2: "No display"
Fix:
  # Make sure you're on the Pi desktop, not SSH
  # Or set DISPLAY:
  export DISPLAY=:0
  python3 main.py

Problem 3: "tkinter not found"
Fix:
  sudo apt-get install python3-tk

Problem 4: "Module not found"
Fix:
  pip3 install -r requirements.txt

Problem 5: Nothing happens when clicking
Fix:
  # Open terminal and run:
  python3 main.py
  # This will show error messages
